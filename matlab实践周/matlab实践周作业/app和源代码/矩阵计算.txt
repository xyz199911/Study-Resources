classdef MatrixCalculator < matlab.apps.AppBase

    % Properties that correspond to app components
    properties (Access = public)
        UIFigure         matlab.ui.Figure
        Panel            matlab.ui.container.Panel
        BtnAdd           matlab.ui.control.Button
        BtnSub           matlab.ui.control.Button
        BtnMul           matlab.ui.control.Button
        BtnDotMul        matlab.ui.control.Button
        BtnTrans         matlab.ui.control.Button
        BtnAnalysis      matlab.ui.control.Button
        BtnSolveEq       matlab.ui.control.Button
        BtnClear         matlab.ui.control.Button
        Panel_2          matlab.ui.container.Panel
        LabelB           matlab.ui.control.Label
        LabelA           matlab.ui.control.Label
        BEditFieldLabel  matlab.ui.control.Label
        BEditField       matlab.ui.control.EditField
        AEditFieldLabel  matlab.ui.control.Label
        AEditField       matlab.ui.control.EditField
        Panel_3          matlab.ui.container.Panel
        Label_3          matlab.ui.control.Label
        ResultTextArea   matlab.ui.control.EditField
    end

methods (Access = private)
function mat = parseInput(app, editComp)
    s = editComp.Value;
    if isempty(s)
        error('输入不能为空，请按MATLAB格式输入，例 [1 2;3 4]');
    end
    mat = str2num(s);
    if isempty(mat)
        error('矩阵格式非法，需用 [] 包裹，分号换行、空格分隔元素');
    end
end
end

    methods (Access = private)

        % Button pushed function: BtnAdd
        function BtnAddButtonPushed(app, event)
        try
        % 解析输入
        A = app.parseInput(app.AEditField);
        B = app.parseInput(app.BEditField);
        % 维度校验
        if ~isequal(size(A),size(B))
            error('加法要求A、B行列维度必须完全一致');
        end
        C = A + B;
        % 成功：淡绿色背景
        app.ResultTextArea.BackgroundColor = [0.8, 1, 0.8];
        app.ResultTextArea.Value = ['A+B = ' mat2str(C,4)];
    catch err
        % 失败：淡红色背景
        app.ResultTextArea.BackgroundColor = [1, 0.8, 0.8];
        app.ResultTextArea.Value = ['运算失败：' err.message];
    end
        
        end

        % Button pushed function: BtnSub
        function BtnSubButtonPushed(app, event)
        try
        A = app.parseInput(app.AEditField);
        B = app.parseInput(app.BEditField);
        if ~isequal(size(A),size(B))
            error('减法要求A、B行列维度必须完全一致');
        end
        C = A - B;
        app.ResultTextArea.BackgroundColor = [0.8, 1, 0.8];
        app.ResultTextArea.Value = ['A-B = ' mat2str(C,4)];
    catch err
        app.ResultTextArea.BackgroundColor = [1, 0.8, 0.8];
        app.ResultTextArea.Value = ['运算失败：' err.message];
    end
       
        end

        % Button pushed function: BtnMul
        function BtnMulButtonPushed(app, event)
       try
        A = app.parseInput(app.AEditField);
        B = app.parseInput(app.BEditField);
        if size(A,2) ~= size(B,1)
            error('矩阵乘法要求：A的列数必须等于B的行数');
        end
        C = A * B;
        app.ResultTextArea.BackgroundColor = [0.8, 1, 0.8];
        app.ResultTextArea.Value = ['A*B = ' mat2str(C,4)];
    catch err
        app.ResultTextArea.BackgroundColor = [1, 0.8, 0.8];
        app.ResultTextArea.Value = ['运算失败：' err.message];
    end
        
        end

        % Button pushed function: BtnDotMul
        function BtnDotMulButtonPushed(app, event)
       try
        A = app.parseInput(app.AEditField);
        B = app.parseInput(app.BEditField);
        if ~isequal(size(A),size(B))
            error('点乘要求A、B行列维度必须完全一致');
        end
        C = A .* B;
        app.ResultTextArea.BackgroundColor = [0.8, 1, 0.8];
        app.ResultTextArea.Value = ['A.*B = ' mat2str(C,4)];
    catch err
        app.ResultTextArea.BackgroundColor = [1, 0.8, 0.8];
        app.ResultTextArea.Value = ['运算失败：' err.message];
       end
        end

        % Button pushed function: BtnTrans
        function BtnTransButtonPushed(app, event)
        
    try
        A = app.parseInput(app.AEditField);
        AT = A';
        app.ResultTextArea.BackgroundColor = [0.8, 1, 0.8];
        app.ResultTextArea.Value = ['A转置 = ' mat2str(AT,4)];
    catch err
        app.ResultTextArea.BackgroundColor = [1, 0.8, 0.8];
        app.ResultTextArea.Value = ['运算失败：' err.message];
    end
        
        end

        % Button pushed function: BtnAnalysis
        function BtnAnalysisButtonPushed(app, event)
       try
        A = app.parseInput(app.AEditField);
        [r,c] = size(A);
        rk = rank(A);
        outStr = sprintf('矩阵维度：%d×%d，矩阵秩 = %d',r,c,rk);
        if r == c
            d = det(A);
            outStr = [outStr sprintf('，行列式 = %.4f',d)];
            if abs(d) > 1e-6
                invA = inv(A);
                outStr = [outStr '，矩阵可逆，逆矩阵 = ' mat2str(invA,4)];
            else
                outStr = [outStr '，矩阵奇异，不可逆'];
            end
        else
            outStr = [outStr '，非方阵，无法计算行列式与逆矩阵'];
        end
        app.ResultTextArea.BackgroundColor = [0.8, 1, 0.8];
        app.ResultTextArea.Value = outStr;
    catch err
        app.ResultTextArea.BackgroundColor = [1, 0.8, 0.8];
        app.ResultTextArea.Value = ['运算失败：' err.message];
       end
        end

        % Button pushed function: BtnSolveEq
        function BtnSolveEqButtonPushed(app, event)
        try
        A = app.parseInput(app.AEditField);
        B = app.parseInput(app.BEditField);
        if size(A,1) ~= size(B,1)
            error('方程组Ax=B要求：A的行数必须等于B的行数');
        end
        x = A \ B;
        res = norm(A*x - B);
        outStr = sprintf('方程组解 x = %s\n残差 = %.6e',mat2str(x,4),res);
        app.ResultTextArea.BackgroundColor = [0.8, 1, 0.8];
        app.ResultTextArea.Value = outStr;
    catch err
        app.ResultTextArea.BackgroundColor = [1, 0.8, 0.8];
        app.ResultTextArea.Value = ['运算失败：' err.message];
        end
        end

        % Button pushed function: BtnClear
        function BtnClearPushed(app, event)
            % 清空两个输入框
    app.AEditField.Value = '';
    app.BEditField.Value = '';
    % 清空结果框 + 恢复默认白色背景
    app.ResultTextArea.Value = '';
    app.ResultTextArea.BackgroundColor = [1, 1, 1];

        end
    end

    methods (Access = public)

        % Construct app
        function app = MatrixCalculator

            % Create and configure components
            createComponents(app)

            % Register the app with App Designer
            registerApp(app, app.UIFigure)

            if nargout == 0
                clear app
            end
        end

        % Code that executes before app deletion
        function delete(app)

            % Delete UIFigure when app is deleted
            delete(app.UIFigure)
        end
    end
